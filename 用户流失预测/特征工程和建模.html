def swich(df):
    cateCols = [c for c in df.columns if df[c].dtype == 'object']
    for col in cateCols:
        if df[col].nunique()==2:
            df[col]=df[col].map(lambda x:1 if x=='Yes'else 0)
        else:
            df=pd.get_dummies(df,columns=[col])
    return df
data=swich(data)
data.to_csv('Processed_data.csv',index=None)

构建模型¶

管道模型pipeline结构：
预处理部分：

  1. 'skb'使用SelectKBest根据卡方统计量筛选特征经过GridSearchCV得到最佳特征数32
  2. 'scaler'再使用StandardScaler(),对数据进行标准化

分类器部分:
由一个投票分类器为主体,其中包括三个分类器,分别为:一个随机森林分类器——'rfr'和两个Bagging分类器'baggtree'和'bagglog'，baggtree是以决策树为基，bagglog是以逻辑回归为基,投票分类方式为soft,
即将所有模型预测样本为某一类别的概率的平均值作为标准，概率最高的对应的类型为最终的预测结果

def GridSearchCV_work(pipeline,train_x,train_y,test_x,test_y,param_grid,score='accuracy_score'):
    gridsearch=GridSearchCV(estimator=pipeline,param_grid=param_grid,cv=3,scoring=score)
    search=gridsearch.fit(train_x,train_y)
    print('最优参数:',search.best_params_)
    print('最优分数: %0.4lf'%search.best_score_)
    predict_y=gridsearch.predict(test_x)
    print(" 准确率 %0.4lf"%accuracy_score(test_y,predict_y))

from sklearn.metrics import accuracy_score
from sklearn.metrics import classification_report,roc_curve,auc,accuracy_score
def model_test_information(model,features_test,labels_test):
    tarin_pred=model.predict(features_train)
    tarin_pro=model.predict_proba(features_train)
    test_pred=model.predict(features_test)
    test_pro=model.predict_proba(features_test)
    fpr1,tpr1,thresholds1= roc_curve(labels_train,tarin_pro[:,1])
    fpr2,tpr2,thresholds2= roc_curve(labels_test,test_pro[:,1])
    plt.figure(figsize=(10,6))
    lw = 2
    roc_auc1 = auc(fpr1,tpr1)
    roc_auc2 = auc(fpr2,tpr2)
    ax1=plt.subplot(121)
    plt.plot(fpr1, tpr1, color='darkorange',lw=lw, 
             label='ROC curve (area = %0.5f)' % roc_auc1)
    plt.plot([0,1], [0,1], color='navy', lw=lw, linestyle='--')
    plt.xlim([0, 1.0])
    plt.ylim([0, 1.05])
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.legend(loc="lower right")
    plt.title('训练集')
    ax2=plt.subplot(122)
    plt.plot(fpr2, tpr2, color='darkorange',lw=lw, 
             label='ROC curve (area = %0.5f)' % roc_auc2)
    plt.plot([0,1], [0,1], color='navy', lw=lw, linestyle='--')
    plt.xlim([0, 1.0])
    plt.ylim([0, 1.05])
    plt.xlabel('False Positive Rate')
    plt.ylabel('True Positive Rate')
    plt.title('测试集')
    plt.suptitle('ROC曲线')
    plt.legend(loc="lower right")
    plt.show()
    print('训练集acc={:4}          测试集acc={}'.format(round(accuracy_score(labels_train,tarin_pred),4),
                                         round(accuracy_score(labels_test,test_pred),4)))
    print('训练集{}'.format(classification_report(labels_train,tarin_pred)))
    print('测试集{}'.format(classification_report(labels_test,test_pred)))

总结¶

经过多次GridSearchCV找到最佳参数后存在过拟合现象，为了解决过拟合我采取了加大逻辑回归的L1正则化系数，和采取bagging的方式成功解决一些过拟合现象，但过拟合仍然存在。可能是由于训练集比较小和特征采集不明显导致的。后续想要提升模型的泛化能力应该扩大训练集量，或者建立新的特征工程。
